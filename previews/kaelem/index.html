Developer API
Streaming completions with deterministic routing

One call fans out to the best model and streams tokens as they land.

const resp = await fetch("https://api.kaelem.dev/v1/route", {
  method: "POST",
  headers: {
    "Authorization": `Bearer ${process.env.KAELEM_KEY}`,
    "X-Tier": "enterprise",
    "X-Latency-Budget": "120"
  },
  body: JSON.stringify({
    session: "ops-demo",
    input: prompt,
    preferred_models: ["gpt-4.1-mini","sonnet-3.5"]
  })
});
for await (const chunk of resp.body) {
  streamToUI(chunk);
}
